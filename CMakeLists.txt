cmake_minimum_required( VERSION 3.15 )

set(CMAKE_POSITION_INDEPENDENT_CODE ON)

project( ensdam VERSION 1.0 LANGUAGES Fortran C )

add_subdirectory(src)

add_dependencies(python fortran wheel)
